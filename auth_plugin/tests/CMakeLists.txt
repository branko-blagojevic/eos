include_directories(${CPPUNIT_INCLUDE_DIRS})

#-------------------------------------------------------------------------------
# EosAuthTests library
#-------------------------------------------------------------------------------
add_library(EosAuthTests MODULE
  AuthFsTest.cc
  Namespace.hh
  TestEnv.cc  TestEnv.hh)

target_link_libraries(EosAuthTests PUBLIC
  XROOTD::CL
  ${CPPUNIT_LIBRARIES})

# Hide Clang deprecated declarations warnings generated by CPPUnit
# Should drop this when once everything is moved to GTest
if (CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  target_compile_options(EosAuthTests PUBLIC -Wno-deprecated-declarations)
endif()

install(TARGETS EosAuthTests
  LIBRARY DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR}
  RUNTIME DESTINATION ${CMAKE_INSTALL_FULL_BINDIR}
  ARCHIVE DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR})
