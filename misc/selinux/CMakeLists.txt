install(DIRECTORY DESTINATION ${CMAKE_INSTALL_DATAROOTDIR}/selinux/targeted)
install(DIRECTORY DESTINATION ${CMAKE_INSTALL_DATAROOTDIR}/selinux/mls)
install(DIRECTORY DESTINATION ${CMAKE_INSTALL_DATAROOTDIR}/selinux/strict)

add_custom_command(OUTPUT ${CMAKE_CURRENT_SOURCE_DIR}/eosfusex.pp
  COMMAND ${CMAKE_CURRENT_SOURCE_DIR}/choose_selinux.sh ${CMAKE_CURRENT_SOURCE_DIR}
  DEPENDS eosfusex.fc eosfusex.te
  COMMENT "Creating eosfusex.pp")

add_custom_target(
  eosfuse-selinux ALL
  DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/eosfusex.pp
  VERBATIM)


install(
  FILES ${CMAKE_CURRENT_SOURCE_DIR}/eosfusex.pp
  DESTINATION ${CMAKE_INSTALL_PREFIX}/${CMAKE_INSTALL_DATAROOTDIR}/selinux/targeted/
  PERMISSIONS OWNER_READ OWNER_EXECUTE
	      GROUP_READ GROUP_EXECUTE
	      WORLD_READ WORLD_EXECUTE)

install(
  FILES ${CMAKE_CURRENT_SOURCE_DIR}/eosfusex.pp
  DESTINATION ${CMAKE_INSTALL_PREFIX}/${CMAKE_INSTALL_DATAROOTDIR}/selinux/mls/
  PERMISSIONS OWNER_READ OWNER_EXECUTE
	      GROUP_READ GROUP_EXECUTE
	      WORLD_READ WORLD_EXECUTE)

install(
  FILES ${CMAKE_CURRENT_SOURCE_DIR}/eosfusex.pp
  DESTINATION ${CMAKE_INSTALL_PREFIX}/${CMAKE_INSTALL_DATAROOTDIR}/selinux/strict/
  PERMISSIONS OWNER_READ OWNER_EXECUTE
	      GROUP_READ GROUP_EXECUTE
	      WORLD_READ WORLD_EXECUTE)
