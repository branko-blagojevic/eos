include_directories(
  ${CMAKE_SOURCE_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR}
  ${CPPUNIT_INCLUDE_DIRS})

link_directories(${CPPUNIT_LIBRARY})

add_library(
  EosFuseTests MODULE
  Namespace.hh
  FuseFsTest.cc
  FuseFileTest.cc
  TestEnv.cc       TestEnv.hh)

target_link_libraries(
  EosFuseTests
  FuseCache
  ${CPPUNIT_LIBRARIES})

# Hide Clang deprecated declarations warnings generated by CPPUnit
# Should drop this when once everything is moved to GTest
if (CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  target_compile_options(EosFuseTests PUBLIC -Wno-deprecated-declarations)
endif()

install(
  TARGETS EosFuseTests
  LIBRARY DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR}
  ARCHIVE DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR})
