include_directories(
  ${CMAKE_SOURCE_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR}
  ${CPPUNIT_INCLUDE_DIRS}
  ${GTEST_INCLUDE_DIRS}
)

link_directories(${CPPUNIT_LIBRARY})

add_library(
  EosFuseTests MODULE
  Namespace.hh
  FuseFsTest.cc
  FuseFileTest.cc
  TestEnv.cc       TestEnv.hh)

add_executable(
  EosFuseCredTests
  ../CredentialFinder.cc
  ../Utils.cc
  ../ProcessInfo.cc
  ../LoginIdentifier.cc
  credential-finder.cc
  login-identifier.cc
  utils.cc
  process-info.cc
)

target_link_libraries(
 EosFuseTests
 FuseCache-Static
 ${CPPUNIT_LIBRARIES})

target_link_libraries(
  EosFuseCredTests
  FuseCache-Static
  gtest_main)

install(
  TARGETS EosFuseTests
  LIBRARY DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR}
  ARCHIVE DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR})
